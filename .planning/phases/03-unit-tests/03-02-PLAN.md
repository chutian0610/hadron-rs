        Router,
        routing::{post, get},
        extract::{Path, State},
        Json,
    };
    use std::sync::Arc;
    use serde::{Deserialize, Serialize};
    use octopus_coordinator::CoordinatorServer;

    #[derive(Clone)]
    pub struct AppState {
        pub coordinator_server: Arc<CoordinatorServer>,
    }

    #[derive(Deserialize)]
    pub struct SubmitQueryRequest {
        pub sql: String,
    }

    #[derive(Serialize)]
    pub struct SubmitQueryResponse {
        pub query_id: String,
    }

    #[derive(Serialize)]
    pub struct QueryStateResponse {
        pub state: String,
    }

    #[derive(Deserialize)]
    pub struct ExplainQueryRequest {
        pub sql: String,
    }

    pub async fn submit_query_handler(...) // extracted
    pub async fn explain_query_handler(...) // extracted
    pub async fn query_state_handler(...) // extracted

    pub fn create_app(coordinator_server: Arc<CoordinatorServer>) -> Router<AppState> {
        Router::new()
            .route("/query/submit", post(submit_query_handler))
            .route("/query/explain", post(explain_query_handler))
            .route("/query/state/:query_id", get(query_state_handler))
            .with_state(AppState { coordinator_server })
    }
